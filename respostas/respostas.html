Questão 01 - SQL

Resposta A:

            
                SELECT
                        p.nome_paciente, a.data_atendimento, a.nome_unidade
                FROM 
                        Pacientes p
                JOIN 
                        Atendimento a ON p.id_paciente = a.id_paciente
                WHERE 
                        a.cd_especialidade = 'Neurologia'
                AND 
                        a.data_atendimento BETWEEN DATE_SUB(CURDATE(), INTERVAL 6  MONTH) AND CURDATE()

Resposta B:

            
                SELECT 
                        a.nome_paciente, b.descricao_prontuario,  c.data_atendimento
                FROM 
                        Prontuario b
                JOIN 
                        Pacientes a ON b.id_paciente = a.id_paciente
                JOIN 
                        Atendimento c ON b.id_atendimento = c.id_atendimento
                WHERE 
                        a.id_paciente = 123;
